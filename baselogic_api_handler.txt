const response = await api.apiPut({
  path: `/product/${req.id}`,
  body: req
})

if (response.status) {
  res.status = response.status
}

res.is_success = response.status! < HttpStatusCode.BadRequest ? true : false

if (response.data) {
  res.data = response.data

  notif.notifSuccess(
    local.localGetFromJson({
      en: `Successfully updated`,
      id: `Berhasil mengubah`
    })
  )
}

if (response.error) {
  res.error = response.error
  error.errorStatusCode(response.status, response.error)
}

if (response.errors) {
  res.errors = response.errors
  notif.notifError(
    local.localGetFromJson({
      en: `Check your form again`,
      id: `Periksa lagi formulir anda`
    })
  )
}
